3. Agent — curl reference (debug)

# Create concept
CONCEPT_ID=$(curl -sf -X POST http://localhost:3000/api/concepts \
  -H 'Content-Type: application/json' \
  -d '{"name":"validate-p3","campaignLabel":"validate"}' | jq -r .id)

# Upload variation image
VAR_BODY=$(curl -sf -X POST "http://localhost:3000/api/concepts/$CONCEPT_ID/variations" \
  -F "file=@e2e/fixtures/sample.png")
VAR_ID=$(echo "$VAR_BODY" | jq -r .id)

# Approve
curl -sf -X PATCH "http://localhost:3000/api/concepts/$CONCEPT_ID/variations/$VAR_ID" \
  -H 'Content-Type: application/json' \
  -d '{"status":"approved"}' | jq .

# List
curl -sf "http://localhost:3000/api/concepts/$CONCEPT_ID/variations" | jq .

# From preview asset (after P2 upload sets preview)
curl -sf -X POST "http://localhost:3000/api/concepts/$CONCEPT_ID/variations/from-asset" \
  -H 'Content-Type: application/json' \
  -d "{\"assetId\":\"ASSET_ID\"}" | jq .

# Serve card rendition
curl -s -o /dev/null -w "HTTP %{http_code}\n" \
  "http://localhost:3000/api/variations/$VAR_ID/file?variant=card"
