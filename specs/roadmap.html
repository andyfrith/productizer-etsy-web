P0 scaffold — commands (excerpt)

pnpm create next-app@latest . --typescript --tailwind --eslint --app --src-dir
pnpm dlx shadcn@latest init
pnpm add drizzle-orm postgres @tanstack/react-query @tanstack/ai zod react-hook-form @hookform/resolvers
pnpm add framer-motion
pnpm add -D drizzle-kit vitest @vitejs/plugin-react jsdom @testing-library/react
pnpm create playwright
docker compose up -d    # Postgres
pnpm drizzle-kit generate && pnpm drizzle-kit migrate
pnpm dev              # Next.js on host
